```python
class DIContainer:
    def __init__(self):
        self.services = {}
        self.factories = {}
        
    def register(self, name, service_class, singleton=True):
        """注册服务"""
        if singleton:
            self.services[name] = service_class()
        else:
            self.factories[name] = service_class
    
    def get(self, name):
        """获取服务实例"""
        if name in self.services:
            return self.services[name]
        elif name in self.factories:
            return self.factories[name]()
        else:
            raise ServiceNotFoundError(f"Service {name} not found")
```

```python
class RecoveryManager:
```
